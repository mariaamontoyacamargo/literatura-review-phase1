\begin{frame}{El problema}{Escala + criterios simultaneos}
\begin{columns}[T]
\column{0.48\textwidth}
\begin{card}
\textbf{\color{bidaccent}La tarea sin agente}\\[3pt]
\scriptsize\color{bidgray}
\begin{itemize}\setlength\itemsep{2pt}
  \item 7 temas (pockets) $\times$ 60 papers = \hl{420 papers}
  \item Leer abstract, calificar metodologia, verificar relevancia
  \item Decidir: Aceptado / Revisar / Rechazado
  \item Extraer sintesis estructurada de los aceptados
  \item Construir red de conexiones entre papers
\end{itemize}
\vspace{2pt}
\textcolor{bidaccent!60}{\footnotesize $\approx$3--4 semanas por investigador}
\end{card}

\column{0.48\textwidth}
\begin{card}
\textbf{\color{bidaccent}Lo que hace el agente}\\[3pt]
\scriptsize\color{bidgray}
\begin{itemize}\setlength\itemsep{2pt}
  \item Busca en \hl{9 fuentes academicas} simultaneamente
  \item Aplica rubrica de 5 dimensiones de forma consistente
  \item Sintetiza findings, metodologia y relevancia para LATAM
  \item Cachea resultados --- segunda corrida cuesta $\approx$\$0
  \item El investigador solo decide los casos \textbf{REVISAR}
\end{itemize}
\vspace{2pt}
\textcolor{bidaccent!60}{\footnotesize Costo total: \textbf{\$5--7 USD} por corrida completa}
\end{card}
\end{columns}

\vspace{3mm}
\begin{card}
\scriptsize\color{bidgray}
\textbf{\color{bidaccent}Contexto:} esta revision informa el diseno del \textbf{primer RCT de adopcion de IA en MiPyMEs de LATAM} (Guatiguara). La literatura debe responder preguntas concretas: que medir, como identificar efectos causales, que heterogeneidades anticipar.
\end{card}
\end{frame}

\begin{frame}{Sistema de scoring v2}{Frameworks de referencia}
\begin{columns}[T]
\column{0.32\textwidth}
\begin{card}
\small\textbf{\color{bidaccent}GRADE}\\[5pt]
\scriptsize\color{bidgray}
Gold standard en medicina y economia del desarrollo.\\[5pt]
Principio clave: separa la \emph{calidad interna} del estudio de su \emph{aplicabilidad} al contexto de politica.\\[5pt]
Informa D1 (jerarquia de evidencia causal) y D3 (transferibilidad).
\end{card}

\column{0.32\textwidth}
\begin{card}
\small\textbf{\color{bidaccent}EPPI-Centre}\\[5pt]
\scriptsize\color{bidgray}
Weight of Evidence, usado en revisiones sistematicas de educacion y ciencias sociales.\\[5pt]
Distingue \emph{soundness of study} (D1--D2) de \emph{relevance of evidence focus} (D4).\\[5pt]
Permite aceptar un buen paper de contexto diferente si la pregunta es la misma.
\end{card}

\column{0.32\textwidth}
\begin{card}
\small\textbf{\color{bidaccent}3ie Evidence Gap Maps}\\[5pt]
\scriptsize\color{bidgray}
Metodologia usada por el BID en evaluaciones de desarrollo.\\[5pt]
Ejes: diseno de identificacion (D1) × relevancia tematica (D4) con celdas vacias = gaps de evidencia.\\[5pt]
Informa la deteccion automatica de gaps en el reporte de Sonnet.
\end{card}
\end{columns}

\vspace{5mm}
\begin{card}
\scriptsize\color{bidgray}
\textbf{\color{bidaccent}Por que 5 dimensiones independientes:} cada dimension mide un aspecto que los demas no pueden sustituir. Un RCT en Silicon Valley puede tener D1=4 y D3=1 (no transferible a LATAM). Un paper de LATAM puede tener D3=3 y D1=1 (solo correlacional). El score final refleja ambas tensiones.
\end{card}
\end{frame}
